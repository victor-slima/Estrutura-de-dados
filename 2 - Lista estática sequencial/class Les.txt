class Les:
    def __init__(self, tamanho):
        self.tam = tamanho
        self.vetor = [None] * self.tam
        self.quant = 0
    
    def show(self):
        for i in range(self.tam):
            print(self.vetor[i], end="")
        print()
    
    def inserir_fim(self, valor):
        self.vetor[self.quant] = valor
        self.quant += 1
    
    def remover_fim(self):
        self.quant -= 1
    
    def esta_cheia(self,)
        if self.quant == self.tam:
            return True
        return False
    
    def esta_vazia(self):
        return self.quant == 0
